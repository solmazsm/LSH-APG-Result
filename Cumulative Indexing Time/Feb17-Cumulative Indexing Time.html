import matplotlib.pyplot as plt


file_path_new = r"C:\Users\Soli1\Desktop\data_LSH_APG\lsh_apg_metrics_audio_v5.csv"

filtered_data_v5 = pd.read_csv(file_path_new, delimiter="\t", engine="python", header=None)


filtered_data_v5.columns = ['Raw']

pattern_all = r"Candidate ID=(\d+).*?InsertionTime=([\d.]+)ns"  # Removed LocalDensity


filtered_data_v5[['Candidate_ID', 'InsertionTime']] = filtered_data_v5['Raw'].str.extract(pattern_all)


filtered_data_v5['InsertionTime'] = pd.to_numeric(filtered_data_v5['InsertionTime'], errors='coerce')

filtered_data_v5.dropna(subset=['InsertionTime'], inplace=True)


print("Number of nodes in filtered_data_v5:", len(filtered_data_v5))


plt.figure(figsize=(10, 6))
plt.plot(range(len(filtered_data_v5)), filtered_data_v5['InsertionTime'].cumsum(), label='v5 Cumulative Time', color='purple')

plt.xlabel('Number of Nodes')
plt.ylabel('Cumulative Indexing Time (ns)')
plt.title('Comparison of Cumulative Indexing Time')
plt.legend()
plt.grid(True)

plt.xlim(0, len(filtered_data_v5))

plt.show()

Number of nodes in filtered_data_v5: 54187
